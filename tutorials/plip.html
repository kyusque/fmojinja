Binding site analysis with Plip¶

https://github.com/pharmai/plip

Execution with docker¶

# export MSYS_NO_PATHCONV=1
docker run --rm -v $(pwd):/results -w /results -u $(id -u ${USER}):$(id -g ${USER}) pharmai/plip -h

%.xml:%.pdb
        export MSYS_NO_PATHCONV=1
        docker run --rm -v $$(pwd):/results -w /results -u $$(id -u $${USER}):$$(id -g $${USER}) pharmai/plip -f $< --dnareceptor -x -O > $@

for i in *.pdb; do make ${i/.pdb/.xml}; done

Read data¶

python -m fmojinja.plip.reader */snapshots/dry/*.xml > data_plip.csv
dos2unix data_plip.csv
